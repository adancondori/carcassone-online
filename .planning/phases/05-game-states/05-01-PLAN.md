  <behavior>
    State transitions:
    - begin_scoring(session, game_id) with playing game -> game.status == "scoring", returns Game
    - begin_scoring(session, game_id) with non-playing game -> raises ValueError
    - begin_scoring(session, 999) -> raises ValueError (game not found)
    - finish_game(session, game_id) with scoring game -> game.status == "finished", returns Game
    - finish_game(session, game_id) with non-scoring game -> raises ValueError
    - finish_game(session, 999) -> raises ValueError (game not found)

    Event-type validation in add_score():
    - add_score(..., event_type="CITY_COMPLETED") in playing state -> succeeds
    - add_score(..., event_type="ROAD_COMPLETED") in playing state -> succeeds
    - add_score(..., event_type="MONASTERY_COMPLETED") in playing state -> succeeds
    - add_score(..., event_type="MANUAL") in playing state -> succeeds
    - add_score(..., event_type="CITY_FINAL") in playing state -> raises ValueError
    - add_score(..., event_type="FARM_FINAL") in playing state -> raises ValueError
    - add_score(..., event_type="CITY_FINAL") in scoring state -> succeeds
    - add_score(..., event_type="ROAD_FINAL") in scoring state -> succeeds
    - add_score(..., event_type="MONASTERY_FINAL") in scoring state -> succeeds
    - add_score(..., event_type="FARM_FINAL") in scoring state -> succeeds
    - add_score(..., event_type="MANUAL") in scoring state -> succeeds
    - add_score(..., event_type="CITY_COMPLETED") in scoring state -> raises ValueError
    - add_score(...) in setup state -> raises ValueError
    - add_score(...) in finished state -> raises ValueError

    Finished-state guards:
    - undo_last(session, game_id) with finished game -> raises ValueError
    - rollback_to(session, game_id, action_id) with finished game -> raises ValueError
    - undo_last(session, game_id) with scoring game -> succeeds (works as before)
    - rollback_to(session, game_id, action_id) with scoring game -> succeeds (works as before)

    Constants:
    - PLAYING_EVENT_TYPES = frozenset({"ROAD_COMPLETED", "CITY_COMPLETED", "MONASTERY_COMPLETED", "MANUAL"})
    - SCORING_EVENT_TYPES = frozenset({"ROAD_FINAL", "CITY_FINAL", "MONASTERY_FINAL", "FARM_FINAL", "MANUAL"})
  </behavior>
